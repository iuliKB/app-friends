```typescript
async function onSubmit() {
  if (!assigneeId) return;
  setPending(true); setErrorText(null);
  const args = kind === 'single'
    ? { groupChannelId, assigneeId, title: title.trim(), isList: false, items: [{ title: title.trim(), due_date: dueDate ? formatLocal(dueDate) : null }] }
```

   ```typescript
   // src/hooks/useGroupMembers.ts (NEW — only if no existing loader)
   import { useCallback, useEffect, useState } from 'react';
   import { supabase } from '@/lib/supabase';

   export interface GroupMember {
     user_id: string;
     display_name: string;
     avatar_url: string | null;
   }

   export function useGroupMembers(groupChannelId: string): {
     members: GroupMember[];
     loading: boolean;
     error: string | null;
   } {
     const [members, setMembers] = useState<GroupMember[]>([]);
     const [loading, setLoading] = useState(true);
     const [error, setError] = useState<string | null>(null);

     const refetch = useCallback(async () => {
       if (!groupChannelId) { setLoading(false); setMembers([]); return; }
       setLoading(true); setError(null);
       const { data, error: err } = await supabase
         .from('group_channel_members')
         .select('user_id, profiles!group_channel_members_user_id_fkey(display_name, avatar_url)')
         .eq('group_channel_id', groupChannelId);
       if (err) { setError(err.message); setMembers([]); }
       else {
         setMembers(((data ?? []) as any[]).map((row) => ({
           user_id: row.user_id,
           display_name: row.profiles?.display_name ?? '',
           avatar_url: row.profiles?.avatar_url ?? null,
         })));
       }
       setLoading(false);
     }, [groupChannelId]);

     useEffect(() => { void refetch(); }, [refetch]);
     return { members, loading, error };
   }
   ```
