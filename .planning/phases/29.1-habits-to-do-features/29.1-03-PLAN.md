```typescript
export function useIOUSummary(): IOUSummaryData {
  const session = useAuthStore((s) => s.session);
  const userId = session?.user?.id ?? null;
  const [rows, setRows] = useState<IOUSummaryRow[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const refetch = useCallback(async () => {
    if (!userId) { setLoading(false); setRows([]); return; }
    setLoading(true); setError(null);
    const { data, error: rpcErr } = await supabase.rpc('get_iou_summary');
    if (rpcErr) { setError(rpcErr.message); setRows([]); }
    else { setRows((data ?? []) as IOUSummaryRow[]); }
    setLoading(false);
  }, [userId]);

  useEffect(() => { refetch(); }, [refetch]);
  return { rows, loading, error, refetch };
}
```

```typescript
function subscribeRealtime(memberIds: string[]) {
  if (channelRef.current) { supabase.removeChannel(channelRef.current); channelRef.current = null; }
  if (memberIds.length === 0) return;
  const filter = `user_id=in.(${memberIds.join(',')})`;
  channelRef.current = supabase
    .channel('user-checkins')
    .on('postgres_changes',
        { event: 'INSERT', schema: 'public', table: 'habit_checkins', filter },
        (_payload) => { void refetch(); })
    .subscribe();
}
```

**File 1: `src/lib/dateLocal.ts`**

```typescript
/**
 * Returns today's date as YYYY-MM-DD in the *device local* timezone.
 *
 * Why this exists: new Date().toISOString().slice(0, 10) returns UTC date,
 * which is off-by-one for users in negative-UTC tzs (Pitfall 5 from RESEARCH.md).
 * We use 'en-CA' locale because it formats as YYYY-MM-DD natively.
 */
export function todayLocal(now: Date = new Date()): string {
  // en-CA formats as YYYY-MM-DD; using {year:'numeric',month:'2-digit',day:'2-digit'}
  // for safety against locale-extension quirks.
  return now.toLocaleDateString('en-CA', {
    year: 'numeric',
    month: '2-digit',
    day: '2-digit',
  });
}
```

```typescript
describe('todayLocal', () => {
  it('returns YYYY-MM-DD shape', () => {
    const result = todayLocal();
    expect(result).toMatch(/^\d{4}-\d{2}-\d{2}$/);
  });

  it('uses provided date when injected', () => {
    // 2026-03-15 in any reasonable tz — date stays the same because we format with locale
    const fixed = new Date('2026-03-15T12:00:00');
    const result = todayLocal(fixed);
    expect(result).toBe('2026-03-15');
  });

  it('uses client local tz, not UTC (Pitfall 5 regression)', () => {
    // 2026-01-01 at 23:30 local should still be 2026-01-01, not 2026-01-02
    // (when local tz is positive-UTC) — and vice versa for negative-UTC.
    // We test by checking that the date roundtrips through local formatting.
    const fixed = new Date(2026, 0, 1, 23, 30, 0);  // local-tz Jan 1 23:30
    const result = todayLocal(fixed);
    expect(result).toBe('2026-01-01');
  });
});
```

```typescript
import { useCallback, useEffect, useRef, useState } from 'react';
import type { RealtimeChannel } from '@supabase/supabase-js';
import { supabase } from '@/lib/supabase';
import { useAuthStore } from '@/store/auth';
import { todayLocal } from '@/lib/dateLocal';
import type { HabitOverviewRow } from '@/types/habits';

export interface UseHabitsResult {
  habits: HabitOverviewRow[];
  loading: boolean;
  error: string | null;
  refetch: () => Promise<void>;
  toggleToday: (habitId: string) => Promise<{ error: string | null }>;
}

export function useHabits(): UseHabitsResult {
  const session = useAuthStore((s) => s.session);
  const userId = session?.user?.id ?? null;
  const [habits, setHabits] = useState<HabitOverviewRow[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  const channelRef = useRef<RealtimeChannel | null>(null);

  const refetch = useCallback(async () => {
    if (!userId) { setLoading(false); setHabits([]); return; }
    setLoading(true); setError(null);
    const { data, error: rpcErr } = await supabase.rpc('get_habits_overview', {
      p_date_local: todayLocal(),
    });
    if (rpcErr) { setError(rpcErr.message); setHabits([]); }
    else { setHabits(((data ?? []) as unknown) as HabitOverviewRow[]); }
    setLoading(false);
  }, [userId]);

  useEffect(() => { void refetch(); }, [refetch]);

  // Realtime — single per-user channel (Pitfall 4)
  useEffect(() => {
    if (!userId || habits.length === 0) return;
    if (channelRef.current) {
      void supabase.removeChannel(channelRef.current);
      channelRef.current = null;
    }
    // Collect every member_id touched by the caller's habits (so we see updates
    // from co-members of group habits). For solo habits, this is just caller.
    // We don't have member_ids in HabitOverviewRow — Realtime can simply filter
    // on user_id IN (<caller>) and rely on Realtime postgres_changes to fire for
    // all caller-relevant rows. Refetch handles re-aggregation for co-members.
    const filter = `user_id=eq.${userId}`;
    channelRef.current = supabase
      .channel(`user-${userId}-checkins`)
      .on('postgres_changes',
        { event: '*', schema: 'public', table: 'habit_checkins', filter },
        () => { void refetch(); })
      .subscribe();
    return () => {
      if (channelRef.current) {
        void supabase.removeChannel(channelRef.current);
        channelRef.current = null;
      }
    };
  }, [userId, habits.length, refetch]);

  const toggleToday = useCallback(async (habitId: string) => {
    // Snapshot+revert pattern (Pitfall 3, Pattern 6)
    let preSnapshot: HabitOverviewRow | null = null;
    setHabits((prev) => {
      preSnapshot = prev.find((h) => h.habit_id === habitId) ?? null;
      return prev;
    });
    if (!preSnapshot) return { error: 'habit-not-found' };

    // Optimistic flip
    setHabits((prev) => prev.map((h) => {
      if (h.habit_id !== habitId) return h;
      const next = !h.did_me_check_in_today;
      return {
        ...h,
        did_me_check_in_today: next,
        completed_today: next ? h.completed_today + 1 : Math.max(0, h.completed_today - 1),
      };
    }));

    const { error: rpcErr } = await supabase.rpc('toggle_habit_today_checkin', {
      p_habit_id: habitId,
      p_date_local: todayLocal(),
    });
    if (rpcErr) {
      // Revert to snapshot
      const snap = preSnapshot;
      setHabits((prev) => prev.map((h) => (h.habit_id === habitId ? snap! : h)));
      return { error: rpcErr.message };
    }
    return { error: null };
  }, []);

  return { habits, loading, error, refetch, toggleToday };
}
```

```typescript
  const session = useAuthStore((s) => s.session);
  const userId = session?.user?.id ?? null;
  const [mine, setMine] = useState<MyTodoRow[]>([]);
  const [fromChats, setFromChats] = useState<ChatTodoRow[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  const refetch = useCallback(async () => {
    if (!userId) { setLoading(false); setMine([]); setFromChats([]); return; }
    setLoading(true); setError(null);
    const today = todayLocal();
    const [{ data: mineData, error: mineErr }, { data: chatData, error: chatErr }] = await Promise.all([
      supabase.rpc('get_my_todos', { p_today: today }),
      supabase.rpc('get_chat_todos', { p_today: today }),
    ]);
    if (mineErr || chatErr) { setError(mineErr?.message ?? chatErr?.message ?? 'unknown'); }
    setMine(((mineData ?? []) as unknown) as MyTodoRow[]);
    setFromChats(((chatData ?? []) as unknown) as ChatTodoRow[]);
    setLoading(false);
  }, [userId]);

  useEffect(() => { void refetch(); }, [refetch]);

  const completeTodo = useCallback(async (todoId: string) => {

    setMine((prev) => { snapshot = prev.find((t) => t.id === todoId) ?? null; return prev; });
    if (!snapshot) return { error: 'todo-not-found' };

    const wasCompleted = snapshot.completed_at !== null;
    const newCompletedAt = wasCompleted ? null : new Date().toISOString();
    setMine((prev) => prev.map((t) => (t.id === todoId ? { ...t, completed_at: newCompletedAt } : t)));

    const { error: updErr } = await supabase
      .from('todos')
      .update({ completed_at: newCompletedAt })
      .eq('id', todoId);
    if (updErr) {
      const snap = snapshot;
      setMine((prev) => prev.map((t) => (t.id === todoId ? snap! : t)));
      return { error: updErr.message };
    }
    return { error: null };
  }, []);

  const completeChatTodo = useCallback(async (itemId: string) => {
    // Optimistically bump done_count for the list containing this item.
    // We don't know list_id from the hook surface; for v1, do RPC then refetch().
    const { data, error: rpcErr } = await supabase.rpc('complete_chat_todo', { p_item_id: itemId });

    void refetch();
```

```typescript
/**
 * Returns true when caller is one missed day away from breaking this habit's streak/target.
 *
 * Deterministic rules per RESEARCH OQ4 (RESOLVED):
 * - Daily: caller has NOT checked in today AND last checkin was yesterday
 *   (missed today after yesterday — streak about to break).
 * - Weekly (cadence='weekly', weekly_target IS NULL → 1× per week):
 *   no completion this week AND today is Fri/Sat/Sun (iso_weekday >= 5).
 * - N×/week: must succeed every remaining day of the week to hit target —
 *   `weekly_target - current_week_completions === remaining_days_left_in_week + 1`
 *   (one more required than days available, so any miss = streak break).
 *
 * Requires `HabitOverviewRow.last_checkin_date_local` + `current_week_completions`
 * fields supplied by Plan 01's `get_habits_overview` RPC.
 */
function isHabitAboutToBreak(habit: HabitOverviewRow, today: Date = new Date()): boolean {
  if (habit.did_me_check_in_today) return false;

  const todayLocalStr = today.toLocaleDateString('en-CA', {
    year: 'numeric', month: '2-digit', day: '2-digit',
  });

  if (habit.cadence === 'daily') {
    if (habit.last_checkin_date_local === null) return false; // never started — not "breaking"
    const yesterday = new Date(today);
    yesterday.setDate(yesterday.getDate() - 1);
    const yesterdayLocalStr = yesterday.toLocaleDateString('en-CA', {
      year: 'numeric', month: '2-digit', day: '2-digit',
    });
    return habit.last_checkin_date_local === yesterdayLocalStr;
  }

  if (habit.cadence === 'weekly') {
    // 1× per week (weekly_target IS NULL); fire on Fri/Sat/Sun if not yet done this week.
    // ISO weekday: Mon=1..Sun=7. Date.getDay(): Sun=0..Sat=6. Convert.
    const dow = today.getDay();
    const isoWeekday = dow === 0 ? 7 : dow;
    return habit.current_week_completions === 0 && isoWeekday >= 5;
  }

  if (habit.cadence === 'n_per_week') {
    if (habit.weekly_target === null) return false; // schema-level guard
    const dow = today.getDay();
    const isoWeekday = dow === 0 ? 7 : dow;
    // Days remaining in week INCLUDING today: 7 - isoWeekday + 1 = 8 - isoWeekday.
    // Days remaining AFTER today: 8 - isoWeekday - 1 = 7 - isoWeekday.
    // The rule says caller must succeed every remaining day INCLUDING today,
    // so "remaining_days_left_in_week" counts today and beyond.
    const remainingDaysIncludingToday = 8 - isoWeekday;
    const remaining = habit.weekly_target - habit.current_week_completions;
    // About to break: must succeed today AND every remaining day (one slip = miss target).
    // Equivalently: remaining === remainingDaysIncludingToday.
    return remaining > 0 && remaining === remainingDaysIncludingToday;
  }

  return false;
}
```
