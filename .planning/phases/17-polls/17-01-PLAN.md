**Exports:**
```typescript
export interface PollOption {
  id: string;
  label: string;
  position: number;
  votes: number;
}

export interface PollState {
  pollId: string;
  question: string;
  options: PollOption[];
  myVotedOptionId: string | null;
  totalVotes: number;
}

export function usePoll(
  pollId: string | null,
  lastPollVoteEvent: { pollId: string; timestamp: number } | null,
): {
  pollState: PollState | null;
  loading: boolean;
  vote: (optionId: string) => Promise<{ error: Error | null }>;
  unVote: () => Promise<{ error: Error | null }>;
}
```

**Auth:** `const session = useAuthStore((s) => s.session); const currentUserId = session?.user?.id ?? '';`

**Null guard:** If `pollId` is null, return `{ pollState: null, loading: false, vote: ..., unVote: ... }` immediately without fetching.

**fetchPollData(pollId: string):**
1. `supabase.from('polls').select('id, question').eq('id', pollId).single()` → pollData
2. `supabase.from('poll_options').select('id, label, position').eq('poll_id', pollId).order('position', { ascending: true })` → options
3. `supabase.from('poll_votes').select('option_id, user_id').eq('poll_id', pollId)` → votes
4. Aggregate: `counts: Record<string, number>` — increment per `option_id` in votes array; set `myVotedOptionId` when `v.user_id === currentUserId`
5. Build `PollState` from pollData, options with counts merged, myVotedOptionId, totalVotes = votes.length
6. Call `setPollState(built)` and `setLoading(false)`

**useEffect:** call `fetchPollData(pollId)` when `pollId` changes (skip if null).

**lastPollVoteEvent useEffect:** When `lastPollVoteEvent` changes AND `lastPollVoteEvent.pollId === pollId`, call `fetchPollData(pollId)` to refresh counts. This bridges useChatRoom Realtime → usePoll without a separate subscription (per D-14).

**vote(optionId: string):**
1. Auth guard: `if (!currentUserId) return { error: new Error('Not authenticated') }`
2. Capture pre-snapshot INSIDE `setPollState` updater (WR-03 pattern): `let preSnapshot: PollState | null = null; setPollState((prev) => { preSnapshot = prev; return prev; });`
3. Apply optimistic update in a second `setPollState` call:
   - If `preSnapshot?.myVotedOptionId === optionId` → this is an un-vote, call `unVote()` instead and return
   - Otherwise: set `myVotedOptionId = optionId`, decrement old option votes (if existed), increment new option votes, recalculate totalVotes
4. DB write — if changing vote (preSnapshot had a myVotedOptionId): `await supabase.from('poll_votes').delete().eq('poll_id', pollId).eq('user_id', currentUserId)`; if that fails, silent revert to preSnapshot and return error
5. `await supabase.from('poll_votes').insert({ poll_id: pollId, option_id: optionId, user_id: currentUserId })`; if that fails, silent revert to preSnapshot and return error
6. Return `{ error: null }`

**unVote():**
1. Auth guard
2. Capture pre-snapshot inside `setPollState` updater (same pattern)
3. Optimistic update: `myVotedOptionId = null`, decrement old option votes, recalculate totalVotes
4. `await supabase.from('poll_votes').delete().eq('poll_id', pollId).eq('user_id', currentUserId)`; silent revert on error
5. Return `{ error: null }`

Note: `vote()` handles both "voting for first time" and "changing vote" — callers just pass the optionId. The `unVote()` path inside `vote()` handles tap-same-option (D-13).

async function sendPoll(question: string, options: string[]): Promise<{ error: Error | null }> {
  if (!currentUserId) return { error: new Error('Not authenticated') };

  const messageId = crypto.randomUUID();
  const tempId = messageId;

  const optimistic: MessageWithProfile = {
    id: tempId,
    plan_id: planId ?? null,
    dm_channel_id: dmChannelId ?? null,
    group_channel_id: groupChannelId ?? null,
    sender_id: currentUserId,
    body: null,
    created_at: new Date().toISOString(),
    image_url: null,
    reply_to_message_id: null,
    message_type: 'poll',
    poll_id: null,   // unknown until RPC completes
    reactions: [],
    pending: true,
    tempId,
    sender_display_name: currentUserDisplayName,
    sender_avatar_url: currentUserAvatarUrl,
  };

  setMessages((prev) => [optimistic, ...prev]);

  // Step 1: insert message row FIRST (create_poll() RPC verifies sender ownership via this row)
  const { error: msgError } = await supabase.from('messages').insert({
    id: messageId,
    plan_id: planId ?? null,
    dm_channel_id: dmChannelId ?? null,
    group_channel_id: groupChannelId ?? null,
    sender_id: currentUserId,
    // eslint-disable-next-line @typescript-eslint/no-explicit-any
    body: null as any,   // Supabase types mark non-nullable; DB nullable since migration 0018
    message_type: 'poll',
  });

  if (msgError) {
    setMessages((prev) => prev.filter((m) => m.tempId !== tempId));
    return { error: msgError };
  }

  // Step 2: atomically create poll + options rows; sets messages.poll_id
  const { error: rpcError } = await supabase.rpc('create_poll', {
    p_message_id: messageId,
    p_question: question,
    p_options: options,
  });

  if (rpcError) {
    setMessages((prev) => prev.filter((m) => m.tempId !== tempId));
    return { error: rpcError };
  }

  return { error: null };
}
```

function handlePollVoteInsert(payload: { new: Record<string, unknown> }) {
  const raw = payload.new;
  const incomingUserId = raw.user_id as string;
  const incomingPollId = raw.poll_id as string;

  // Dedup: own INSERT already applied via optimistic update in usePoll
  if (incomingUserId === currentUserId) return;

  // Scope guard: verify poll_id belongs to a message in current room
  setMessages((prev) => {
    const msgIdx = prev.findIndex((m) => m.poll_id === incomingPollId);
    if (msgIdx === -1) return prev; // not in this room
    return prev; // no messages state change — just signal PollCard via lastPollVoteEvent
  });

  // Signal PollCard to refresh counts (bridge to usePoll without duplicate subscription)
  setLastPollVoteEvent({ pollId: incomingPollId, timestamp: Date.now() });
}

function handlePollVoteDelete(payload: { old: Record<string, unknown> }) {
  const raw = payload.old;
  const deletedUserId = raw.user_id as string;
  const incomingPollId = raw.poll_id as string;

  if (deletedUserId === currentUserId) return;

  setMessages((prev) => {
    const msgIdx = prev.findIndex((m) => m.poll_id === incomingPollId);
    if (msgIdx === -1) return prev;
    return prev;
  });

  setLastPollVoteEvent({ pollId: incomingPollId, timestamp: Date.now() });
}
```
