```typescript
interface PollCreationSheetProps {
```

**State:**

```typescript
const [question, setQuestion] = useState('');

```

**Animation — copy SendBar pattern verbatim:**
- `open()`: `Animated.timing(translateY, { toValue: 0, duration: 250, useNativeDriver: true }).start()`
- `close()`: `Animated.timing(translateY, { toValue: 300, duration: 200, useNativeDriver: true }).start(() => onDismiss())`
- `useEffect(() => { if (visible) { open(); } }, [visible])` — triggers open animation when sheet becomes visible

**Reset state on dismiss:** After `onDismiss()` is called, reset `question` and `options` back to `['', '']` so next open is fresh. Reset inside the `onDismiss` callback or after animation completes.

**canSendPoll derived value:**
```typescript
const canSendPoll = question.trim().length > 0 && options.every((o) => o.trim().length > 0);
```

**handleSend:**
```typescript
function handleSend() {
  if (!canSendPoll) return;
  onSend(question.trim(), options.map((o) => o.trim()));
  close();  // triggers animation then onDismiss
}
```

**addOption:** `setOptions((prev) => [...prev, ''])` — only callable when `options.length < 4`

**removeOption(index):** `setOptions((prev) => prev.filter((_, i) => i !== index))` — only rendered for index >= 2 (options 3 and 4)

**JSX structure:**
```jsx
<Modal visible={visible} transparent animationType="none" onRequestClose={close}>
  <TouchableWithoutFeedback onPress={close}>
    <View style={styles.backdrop} />
  </TouchableWithoutFeedback>
  <KeyboardAvoidingView behavior={Platform.OS === 'ios' ? 'padding' : undefined} style={styles.kavContainer}>
    <Animated.View style={[styles.sheet, { transform: [{ translateY }] }]}>
      <View style={styles.dragHandle} />
      {/* Header: "New Poll" */}
      <Text style={styles.header}>New Poll</Text>
      {/* Question input */}
      <TextInput
        style={styles.input}
        value={question}
        onChangeText={setQuestion}
        placeholder="Ask the group…"
        placeholderTextColor={COLORS.text.secondary}
        returnKeyType="next"
      />
      {/* Option rows */}
      {options.map((opt, idx) => (
        <View key={idx} style={styles.optionRow}>
          <TextInput
            style={[styles.input, styles.optionInput]}
            value={opt}
            onChangeText={(text) => setOptions((prev) => prev.map((o, i) => i === idx ? text : o))}
            placeholder={`Option ${idx + 1}`}
            placeholderTextColor={COLORS.text.secondary}
            returnKeyType="next"
          />
          {idx >= 2 && (
            <TouchableOpacity
              onPress={() => removeOption(idx)}
              accessibilityLabel={`Remove option ${idx + 1}`}
              style={styles.removeButton}
            >
              <Ionicons name="close-circle-outline" size={20} color={COLORS.text.secondary} />
            </TouchableOpacity>
          )}
        </View>
      ))}
      {/* + Add option — hidden when 4 options */}
      {options.length < 4 && (
        <TouchableOpacity
          onPress={addOption}
          style={styles.addOptionRow}
          accessibilityLabel="Add another option"
        >
          <Ionicons name="add" size={20} color={COLORS.interactive.accent} />
          <Text style={styles.addOptionLabel}>+ Add option</Text>
        </TouchableOpacity>
      )}
      {/* Button row */}
      <View style={styles.buttonRow}>
        <TouchableOpacity onPress={close} style={styles.discardButton}>
          <Text style={styles.discardLabel}>Discard Poll</Text>
        </TouchableOpacity>
        <TouchableOpacity
          onPress={handleSend}
          disabled={!canSendPoll}
          style={[styles.sendButton, !canSendPoll && styles.sendButtonDisabled]}
          accessibilityLabel={canSendPoll ? 'Send Poll' : 'Send Poll, disabled — fill in question and all options'}
        >
          <Text style={[styles.sendLabel, !canSendPoll && styles.sendLabelDisabled]}>Send Poll</Text>
        </TouchableOpacity>
      </View>
    </Animated.View>
  </KeyboardAvoidingView>
</Modal>
```
