    import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query';
    import { supabase } from '@/lib/supabase';
    import { useAuthStore } from '@/stores/useAuthStore';
    import { useStatusStore } from '@/stores/useStatusStore';
    import { queryKeys } from '@/lib/queryKeys';
    // <import notification helpers as before — those stay>

    export function useStatus() {
      const session = useAuthStore((s) => s.session);
      const userId = session?.user?.id ?? null;
      const queryClient = useQueryClient();
      const key = queryKeys.status.own(userId ?? '');

      const query = useQuery({
        queryKey: key,
        queryFn: async () => {
          const { data, error } = await supabase
            .from('effective_status')
            .select('*')
            .eq('user_id', userId!)
            .maybeSingle();
          if (error) throw error;
          return data;
        },
        enabled: !!userId,
      });

      // Mirror query result into useStatusStore for outside-React reads (notification dispatcher).
      // This is the load-bearing line that keeps the dispatcher working — research Open Q #3.
      useEffect(() => {
        if (query.data !== undefined) {
          useStatusStore.getState().setCurrentStatus(query.data);
        }
      }, [query.data]);

      const setMutation = useMutation({
        mutationFn: async (next: CurrentStatus) => {
          const { error } = await supabase.from('statuses').upsert(/* next row */);
          if (error) throw error;
        },
        onMutate: async (next) => {
          await queryClient.cancelQueries({ queryKey: key });
          const previous = queryClient.getQueryData(key);
          queryClient.setQueryData(key, next);
          // ALSO mirror to useStatusStore (outside-React read path)
          useStatusStore.getState().setCurrentStatus(next);
          return { previous };
        },
        onError: (_err, _input, ctx) => {
          if (ctx?.previous) {
            queryClient.setQueryData(key, ctx.previous);
            useStatusStore.getState().setCurrentStatus(ctx.previous as any);
          }
        },
        onSettled: () => {
          void queryClient.invalidateQueries({ queryKey: key });
          void queryClient.invalidateQueries({ queryKey: queryKeys.home.friends(userId ?? '') });
        },
      });

      // <preserve touch() and any other public-shape methods from pre-migration file>

      return {
        status: query.data ?? null,
        loading: query.isLoading,
        error: query.error ? (query.error as Error).message : null,
        setStatus: async (next: CurrentStatus) => {
          try { await setMutation.mutateAsync(next); return { error: null }; }
          catch (err) { return { error: err instanceof Error ? err.message : 'setStatus failed' }; }
        },
        // <preserve touch / refresh / other public methods from pre-migration>
      };
    }
