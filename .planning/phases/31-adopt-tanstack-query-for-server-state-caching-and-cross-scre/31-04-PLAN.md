    import { queryKeys } from '@/lib/queryKeys';
    import type { PlanRow, PlanMember /* etc */ } from '@/types/plans'; // read pre-migration file for actual types

    export function usePlans() {
      const session = useAuthStore((s) => s.session);
      const userId = session?.user?.id ?? null;
      const queryClient = useQueryClient();
      const listKey = queryKeys.plans.list(userId ?? '');

      const query = useQuery({
        queryKey: listKey,
        queryFn: async (): Promise<PlanWithMembers[]> => {
          // <copy the multi-step join body from the pre-migration file verbatim;
          //  replace setState with `return joined` and the error path with `throw error`>
        },
        enabled: !!userId,
      });

      const rsvpMutation = useMutation({
        mutationFn: async (input: { planId: string; status: 'yes' | 'no' | 'maybe' }) => {
          const { error } = await supabase.rpc('rsvp_plan', { p_plan_id: input.planId, p_status: input.status });
          if (error) throw error;
        },
        onMutate: async (input) => {
          await queryClient.cancelQueries({ queryKey: listKey });
          await queryClient.cancelQueries({ queryKey: queryKeys.plans.detail(input.planId) });
          const previousList = queryClient.getQueryData<PlanWithMembers[]>(listKey);
          const previousDetail = queryClient.getQueryData(queryKeys.plans.detail(input.planId));
          queryClient.setQueryData<PlanWithMembers[]>(listKey, (old) =>
            old?.map((p) => p.id === input.planId
              ? { ...p, members: p.members?.map((m) => m.user_id === userId ? { ...m, rsvp_status: input.status } : m) ?? [] }
              : p,
            ) ?? [],
          );
          // similar setQueryData for plans.detail(input.planId) — read pre-migration file for shape
          return { previousList, previousDetail };
        },
        onError: (_err, input, ctx) => {
          if (ctx?.previousList) queryClient.setQueryData(listKey, ctx.previousList);
          if (ctx?.previousDetail) queryClient.setQueryData(queryKeys.plans.detail(input.planId), ctx.previousDetail);
        },
        onSettled: (_data, _err, input) => {
          void queryClient.invalidateQueries({ queryKey: listKey });
          void queryClient.invalidateQueries({ queryKey: queryKeys.plans.detail(input.planId) });
          void queryClient.invalidateQueries({ queryKey: queryKeys.home.upcomingEvents(userId ?? '') });
        },
      });

      // @mutationShape: no-optimistic
      const createMutation = useMutation({
        mutationFn: async (input: PlanCreateInput) => {
          const { data, error } = await supabase.rpc('create_plan', { /* args from pre-migration file */ });
          if (error) throw error;
          return data;
        },
        onSuccess: () => {
          void queryClient.invalidateQueries({ queryKey: listKey });
          void queryClient.invalidateQueries({ queryKey: queryKeys.home.upcomingEvents(userId ?? '') });
        },
      });

      return {
        plans: query.data ?? [],
        loading: query.isLoading,
        error: query.error ? (query.error as Error).message : null,
        refetch: query.refetch,
        rsvp: async (planId: string, status: 'yes' | 'no' | 'maybe') => {
          try { await rsvpMutation.mutateAsync({ planId, status }); return { error: null }; }
          catch (err) { return { error: err instanceof Error ? err.message : 'rsvp failed' }; }

          catch (err) { return { data: null, error: err instanceof Error ? err.message : 'create failed' }; }
        },
      };
    }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| signed-URL contents (1h TTL) → query cache | Wave 8 persistence will exclude this key family to avoid serving expired URLs on cold start. Wave 4 leaves the in-memory cache as-is. |
| RSVP mutation payload → server | Server RLS enforces that the caller can only RSVP for themselves. Optimistic write only mutates the caller's own member row. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-31-13 | Information Disclosure | Plan-photos signed URLs survive sign-out | mitigate | Wave 1 authBridge.removeQueries clears all keys, including `plans.photos(planId)` and `plans.allPhotos(userId)`. Wave 8 persistence config excludes both via `shouldDehydrateQuery`. |
| T-31-14 | Tampering | RSVP optimistic write affects other members' rows | mitigate | The setQueryData callback only updates `m.user_id === userId` — defends against accidental cross-member writes. Server RLS is the authoritative gate. |
| T-31-15 | Repudiation | Plan create succeeds on server but client thinks it failed (or vice versa) | mitigate | createPlan uses `onSuccess: invalidate` (no optimistic). The mutation's `mutateAsync` resolves only when the server confirms; failures surface via `{data: null, error}` return value. |
</threat_model>
