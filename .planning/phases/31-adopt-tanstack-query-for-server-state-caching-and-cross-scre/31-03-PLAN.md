```ts
export function subscribeHomeStatuses(
  queryClient: QueryClient,
  userId: string,
  friendIds: string[],
): () => void {
  const channelName = `home-statuses-${userId}`; // userId scope, NOT friendIds (refCount across renders)
  const existing = registry.get(channelName);
  if (existing) { existing.refCount++; return () => releaseSubscription(channelName); }

  const filter = friendIds.length > 0 ? `user_id=in.(${friendIds.join(',')})` : '';
  const channel = supabase.channel(channelName)
    .on('postgres_changes', { event: '*', schema: 'public', table: 'statuses', filter }, () => {
      void queryClient.invalidateQueries({ queryKey: queryKeys.home.friends(userId) });
    })
    .subscribe();

  registry.set(channelName, { refCount: 1, teardown: () => { void supabase.removeChannel(channel); } });
  return () => releaseSubscription(channelName);
}
```

    ```ts
    /** Subscribe to statuses changes for the caller's friend set. Returns unsubscribe.
     * Channel name is scoped by userId (not friendIds) so refCount works across renders
     * where friendIds changes (a friend being added/removed only invalidates the friends
     * list itself, not this subscription's cache key).
     */
    export function subscribeHomeStatuses(
      queryClient: QueryClient,
      userId: string,
      friendIds: string[],
    ): Unsubscribe {
      if (friendIds.length === 0) return () => {};

      const channelName = `home-statuses-${userId}`;
      const existing = registry.get(channelName);
      if (existing) {
        existing.refCount++;
        return () => releaseSubscription(channelName);
      }

      const filter = `user_id=in.(${friendIds.join(',')})`;
      const channel = supabase
        .channel(channelName)
        .on(
          'postgres_changes',
          { event: '*', schema: 'public', table: 'statuses', filter },
          (_payload) => {
            void queryClient.invalidateQueries({ queryKey: queryKeys.home.friends(userId) });
          },
        )
        .subscribe();

      registry.set(channelName, {
        refCount: 1,
        teardown: () => { void supabase.removeChannel(channel); },
      });

      return () => releaseSubscription(channelName);
    }
    ```

    ```ts
    import { supabase } from '@/lib/supabase';
    import { useAuthStore } from '@/stores/useAuthStore';
    import { todayLocal } from '@/lib/dateLocal';
    import { queryKeys } from '@/lib/queryKeys';
    import type { MyTodoRow } from '@/types/todos';

    export function useTodos() {
      const session = useAuthStore((s) => s.session);
      const userId = session?.user?.id ?? null;
      const today = todayLocal();
      const queryClient = useQueryClient();
      const key = queryKeys.todos.mine(today);

      const query = useQuery<MyTodoRow[]>({
        queryKey: key,
        queryFn: async () => {
          const { data, error } = await (supabase as any).rpc('get_my_todos', { p_date_local: today });
          if (error) throw error;
          return (data ?? []) as MyTodoRow[];
        },
        enabled: !!userId,
      });

      const addMutation = useMutation({
        mutationFn: async (input: { title: string; due_date?: string | null }) => {
          const { data, error } = await (supabase as any).rpc('add_my_todo', {
            p_title: input.title,
            p_due_date: input.due_date ?? null,
          });
          if (error) throw error;
          return data as MyTodoRow;
        },
        onMutate: async (input) => {
          await queryClient.cancelQueries({ queryKey: key });
          const previous = queryClient.getQueryData<MyTodoRow[]>(key);
          // Hermes lacks crypto.randomUUID — use the Math.random template (STATE.md decision)
          const tempId = `tmp_${Math.random().toString(36).slice(2)}_${Date.now()}`;
          const optimistic: MyTodoRow = {
            id: tempId,
            title: input.title,
            due_date: input.due_date ?? null,
            completed_at: null,
            // <plus any other required MyTodoRow fields with safe defaults>
          } as MyTodoRow;
          queryClient.setQueryData<MyTodoRow[]>(key, (old) => [optimistic, ...(old ?? [])]);
          return { previous };
        },
        onError: (_err, _input, ctx) => {
          if (ctx?.previous) queryClient.setQueryData(key, ctx.previous);
        },
        onSettled: () => {
          void queryClient.invalidateQueries({ queryKey: key });
          void queryClient.invalidateQueries({ queryKey: queryKeys.home.all() });
        },
      });

      const toggleMutation = useMutation({
        mutationFn: async (id: string) => {
          const { error } = await (supabase as any).rpc('toggle_my_todo', { p_todo_id: id });
          if (error) throw error;
        },
        onMutate: async (id) => {
          await queryClient.cancelQueries({ queryKey: key });
          const previous = queryClient.getQueryData<MyTodoRow[]>(key);
          queryClient.setQueryData<MyTodoRow[]>(key, (old) =>
            old?.map((t) =>
              t.id === id
                ? { ...t, completed_at: t.completed_at ? null : new Date().toISOString() }
                : t,
            ) ?? [],
          );
          return { previous };
        },
        onError: (_err, _id, ctx) => {
          if (ctx?.previous) queryClient.setQueryData(key, ctx.previous);
        },
        onSettled: () => {
          void queryClient.invalidateQueries({ queryKey: key });
          void queryClient.invalidateQueries({ queryKey: queryKeys.home.all() });
        },
      });

      const deleteMutation = useMutation({
        mutationFn: async (id: string) => {
          const { error } = await (supabase as any).rpc('delete_my_todo', { p_todo_id: id });
          if (error) throw error;
        },
        onMutate: async (id) => {

          const previous = queryClient.getQueryData<MyTodoRow[]>(key);
          queryClient.setQueryData<MyTodoRow[]>(key, (old) => old?.filter((t) => t.id !== id) ?? []);

        },

        },
        onSettled: () => {

          void queryClient.invalidateQueries({ queryKey: queryKeys.home.all() });
        },
    ```
