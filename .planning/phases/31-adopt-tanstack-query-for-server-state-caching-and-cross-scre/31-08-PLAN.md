```ts
useMutation({
  mutationFn: async (input: { body: string; channelId: string }) => { /* supabase insert */ },
  onMutate: async (input) => {
    await queryClient.cancelQueries({ queryKey: queryKeys.chat.messages(input.channelId) });
    const previous = queryClient.getQueryData(queryKeys.chat.messages(input.channelId));
    const tempId = `tmp_${Math.random().toString(36).slice(2)}_${Date.now()}`;
    queryClient.setQueryData(queryKeys.chat.messages(input.channelId), (old: any[]) => [
      { id: tempId, ...input, pending: true, failed: false },
      ...(old ?? []),
    ]);
    return { previous, tempId };
  },
  onError: (_err, input, ctx) => {
    // Mark the optimistic as failed (CHAT-03 pattern) rather than removing it
    queryClient.setQueryData(queryKeys.chat.messages(input.channelId), (old: any[]) =>
      old?.map((m) => (ctx?.tempId && m.id === ctx.tempId ? { ...m, pending: false, failed: true } : m)) ?? [],
    );
  },
  onSettled: () => { /* NONE — Realtime INSERT will reconcile (research §Pattern 5 line 493) */ },
});
```

Note for chat optimistic: `onSettled` is intentionally a no-op (or void) because the Realtime INSERT replay performs reconciliation. To still pass the mutationShape gate, we use the `@mutationShape: no-optimistic` marker — NO, that's wrong: the mutation IS optimistic. The mutationShape regression test currently requires onSettled. Solution: add a no-op `onSettled: () => {}` so the gate sees all four members. (Real reconciliation happens in subscribeChatRoom on INSERT.) The marker is reserved for plain non-optimistic mutations only.

    ```ts
    /** Subscribe to chat_room messages for a given channelId. Hybrid:
     *    INSERT → setQueryData prepend with id-dedup (handles own-user optimistic replay)
     *    UPDATE → invalidateQueries (REPLICA IDENTITY default ships PK-only payloads)
     *    DELETE → setQueryData filter by id
     */
    export function subscribeChatRoom(
      queryClient: QueryClient,
      channelId: string,
    ): Unsubscribe {
      const channelName = `chat-${channelId}`;
      const existing = registry.get(channelName);
      if (existing) {
        existing.refCount++;
        return () => releaseSubscription(channelName);
      }

      const channel = supabase
        .channel(channelName)
        .on(
          'postgres_changes',
          { event: 'INSERT', schema: 'public', table: 'messages', filter: `channel_id=eq.${channelId}` },
          (payload) => {
            const incoming = payload.new as { id: string } & Record<string, unknown>;
            if (!incoming?.id) return;
            queryClient.setQueryData<any[]>(queryKeys.chat.messages(channelId), (old) => {
              if (!old) return old;
              // 1) Optimistic replacement: same client-generated id flagged pending
              const optimisticIdx = old.findIndex((m: any) => m.pending && m.id === incoming.id);
              if (optimisticIdx >= 0) {
                const next = [...old];
                next[optimisticIdx] = { ...incoming, pending: false, failed: false } as any;
                return next;
              }
              // 2) Server already mirrors this id — no-op dedup
              if (old.some((m: any) => m.id === incoming.id)) return old;
              // 3) New row
              return [incoming, ...old];
            });
          },
        )
        .on(
          'postgres_changes',
          { event: 'UPDATE', schema: 'public', table: 'messages', filter: `channel_id=eq.${channelId}` },
          () => {
            void queryClient.invalidateQueries({ queryKey: queryKeys.chat.messages(channelId) });
          },
        )
        .on(
          'postgres_changes',
          { event: 'DELETE', schema: 'public', table: 'messages', filter: `channel_id=eq.${channelId}` },
          (payload) => {
            const removed = payload.old as { id?: string } | null;
            if (!removed?.id) return;
            queryClient.setQueryData<any[]>(queryKeys.chat.messages(channelId), (old) =>
              old?.filter((m: any) => m.id !== removed.id) ?? [],
            );
          },
        )
        .subscribe();

      registry.set(channelName, {
        refCount: 1,
        teardown: () => { void supabase.removeChannel(channel); },
      });

      return () => releaseSubscription(channelName);
    }
    ```
