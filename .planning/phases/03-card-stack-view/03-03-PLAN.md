**handleSkip (D-11 — auto-loop):**
```typescript
function handleSkip() {
  setCurrentIndex(prev => {
    if (deck.length === 0) return 0;
    return (prev + 1) % deck.length; // modulo wraps at end
  });
  // Record dismissed friend for undo (D-10)
  if (deck.length > 0) {
    setDismissedStack(prev => [...prev, deck[currentIndex]]);
  }
}
```

**handleUndo (D-10):**
```typescript
function handleUndo() {
  if (dismissedStack.length === 0) return;
  setDismissedStack(prev => prev.slice(0, -1));
  setCurrentIndex(prev => {
    if (deck.length === 0) return 0;
    return (prev - 1 + deck.length) % deck.length;
  });
}
```

**Stack rendering (D-06 — UI-SPEC StackDepthEffect values):**
```typescript
// Render up to 3 cards. Front card = FriendSwipeCard (keyed by friend_id).
// Background cards = plain Views with COLORS.surface.card fill (no gradient).
// Use absolute positioning for the stack container.
// STACK_CONFIGS[0] = front (zIndex 30), [1] = depth-1 (zIndex 20), [2] = depth-2 (zIndex 10)

const STACK_CONFIGS = [
  // eslint-disable-next-line campfire/no-hardcoded-styles
  { translateY: 0,  scale: 1.0,  zIndex: 30, opacity: 1.0  },
  // eslint-disable-next-line campfire/no-hardcoded-styles
  { translateY: 8,  scale: 0.95, zIndex: 20, opacity: 0.6  },
  // eslint-disable-next-line campfire/no-hardcoded-styles
  { translateY: 16, scale: 0.90, zIndex: 10, opacity: 0.35 },
] as const;

// Render in REVERSE order (lowest z-index first so front card is on top):
const cardsToRender = Math.min(STACK_CONFIGS.length, deck.length);

// In JSX:
<View style={styles.stackContainer}>
  {STACK_CONFIGS.slice(0, cardsToRender).reverse().map((config, reversedPos) => {
    const stackPos = cardsToRender - 1 - reversedPos; // back to forward index
    const friendIndex = (currentIndex + stackPos) % deck.length;
    const friend = deck[friendIndex];

    if (stackPos === 0) {
      // Front card — animated FriendSwipeCard, keyed by friend_id for fresh state on each advance
      return (
        <FriendSwipeCard
          key={friend.friend_id}
          friend={friend}
          onSkip={handleSkip}
          onUndo={handleUndo}
          width={CARD_WIDTH}
          style={{ zIndex: config.zIndex }}
        />
      );
    }

    // Background depth cards — plain View, no gesture, no gradient
    return (
      <View
        key={`depth-${stackPos}-${friend.friend_id}`}
        style={[
          styles.depthCard,
          {
            width: CARD_WIDTH,
            zIndex: config.zIndex,
            opacity: config.opacity,
            transform: [
              { translateY: config.translateY },
              { scale: config.scale },
            ],
          },
        ]}
      />
    );
  })}
</View>
```
