Add `formatTurningAge` after the existing `formatBirthdayDate` function. Do NOT modify existing functions:
```typescript
/**
 * Returns "turning N" string for the age the person will be on their next birthday.
 * Call only when birthday_year !== null.
 * Feb 29 guard: clamps Feb 29 to Feb 28 in non-leap reference years.
 */
export function formatTurningAge(
  year: number,
  month: number,
  day: number,
  referenceDate: Date = new Date()
): string {
  // Leap-year guard: clamp Feb 29 to Feb 28 in non-leap reference years
  const safeDay =
    month === 2 && day === 29 && referenceDate.getFullYear() % 4 !== 0 ? 28 : day;
  const thisYearBirthday = new Date(referenceDate.getFullYear(), month - 1, safeDay);
  const turningYear =
    thisYearBirthday >= referenceDate
      ? referenceDate.getFullYear()
      : referenceDate.getFullYear() + 1;
  return `turning ${turningYear - year}`;
}
```
