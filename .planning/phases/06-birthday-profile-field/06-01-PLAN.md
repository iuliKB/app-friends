**Props interface (exact):**
```typescript
interface BirthdayPickerProps {
  month: number | null;   // 1-12 or null
  day: number | null;     // 1-31 or null
  onChange: (month: number | null, day: number | null) => void;
  disabled?: boolean;
}
```

**Data constants (define at module level, outside component):**
```typescript
const MONTH_NAMES = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
const MONTH_NAMES_FULL = ['January', 'February', 'March', 'April', 'May', 'June', 'July', 'August', 'September', 'October', 'November', 'December'];

function getDaysInMonth(month: number): number {
  const days: Record<number, number> = {
    1: 31, 2: 29, 3: 31, 4: 30, 5: 31, 6: 30,
    7: 31, 8: 31, 9: 30, 10: 31, 11: 30, 12: 31,
  };
  return days[month] ?? 31;
}
```

Note: Feb returns 29 (the DB CHECK allows 1-29 for month=2). Feb 29 → Feb 28 normalization happens only in handleSave in edit.tsx (Plan 02), not here.
