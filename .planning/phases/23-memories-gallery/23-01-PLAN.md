**deletePhoto function** — cross-plan variant (takes planId to locate photo in state):
```typescript
const deletePhoto = useCallback(
  async (photoId: string, planId: string): Promise<{ error: Error | null }> => {
    if (!session?.user) return { error: new Error('Not authenticated') };
    const group = groups.find((g) => g.planId === planId);
    const photo = group?.photos.find((p) => p.id === photoId);
    if (!photo) return { error: new Error('Photo not found in local state') };

    const { error: dbError } = await supabase
      .from('plan_photos')
      .delete()
      .eq('id', photoId);
    if (dbError) return { error: new Error(dbError.message) };

    const { error: storageError } = await supabase.storage
      .from('plan-gallery')
      .remove([(photo as { storagePath: string }).storagePath]);
    if (storageError) {
      console.error('[useAllPlanPhotos] Storage delete failed (row already deleted):', storageError.message);
    }
    await refetch();
    return { error: null };
  },
  [groups, session?.user, refetch]
);
```
