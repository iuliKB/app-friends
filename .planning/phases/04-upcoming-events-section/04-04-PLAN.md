**1. Create src/lib/uploadPlanCover.ts:**

```typescript
import { supabase } from '@/lib/supabase';

/**
 * Uploads a local image URI to Supabase Storage plan-covers bucket.
 * D-15: Uses Supabase Storage for image hosting.
 * Returns the public URL on success, or null on failure.
 *
 * Pattern: fetch(localUri) → blob → arrayBuffer (avoids base64-arraybuffer dep)
 * upsert: true allows safe re-upload when editing an existing plan's cover.
 */
export async function uploadPlanCover(
  planId: string,
  localUri: string
): Promise<string | null> {
  try {
    const response = await fetch(localUri);
    const blob = await response.blob();
    const arrayBuffer = await blob.arrayBuffer();

    const path = `${planId}/cover.jpg`;

    const { error: uploadError } = await supabase.storage
      .from('plan-covers')
      .upload(path, arrayBuffer, {
        contentType: 'image/jpeg',
        upsert: true, // allow re-upload on plan edit
      });

    if (uploadError) {
      console.error('[uploadPlanCover] Upload failed:', uploadError.message);
      return null;
    }

    const { data } = supabase.storage.from('plan-covers').getPublicUrl(path);
    return data.publicUrl;
  } catch (err) {
    const message = err instanceof Error ? err.message : String(err);
    console.error('[uploadPlanCover] Unexpected error:', message);
    return null;
  }
}
```

|----------|-------------|
| User device (camera roll) → Supabase Storage | Local file URI crosses into network upload |
| Supabase Storage → EventCard | Public URL retrieved from Storage and rendered in expo-image |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-08 | Tampering | uploadPlanCover path | mitigate | Storage path is `${planId}/cover.jpg` where planId is a Supabase UUID — no user-controlled path segments (RESEARCH.md security domain: prevent path traversal) |
| T-04-09 | Tampering | Unauthorized cover overwrite | mitigate | PlanDashboardScreen guards cover edit behind `session.user.id === plan.created_by` check; Supabase Storage INSERT/UPDATE policy should restrict to plan creator (configure in Supabase dashboard: policy `auth.uid() = (SELECT created_by FROM plans WHERE id = [planId])`) |
| T-04-10 | Tampering | Arbitrary file upload | mitigate | `expo-image-picker mediaTypes: ['images']` restricts at OS level; `contentType: 'image/jpeg'` set on upload; additional MIME validation via Storage bucket content-type policy is recommended |
| T-04-11 | Information Disclosure | Public cover image URLs | accept | Cover images are intentionally public (social feature). If plan is deleted, Storage object cleanup should be added to deletePlan in usePlanDetail (note for future phase) |
| T-04-12 | Denial of Service | Large image upload | mitigate | `quality: 0.8` in launchImageLibraryAsync compresses before upload; expo-image-picker enforces device photo dimensions |
</threat_model>
