<!-- Scale spring pattern — replicate across all 4 card components -->
```typescript
// Required imports (add if not present):
import React, { useMemo, useRef } from 'react';
import { Animated, ... } from 'react-native';
import { ANIMATION } from '@/theme';

// Inside component function body:
const scaleAnim = useRef(new Animated.Value(1)).current;

function handlePressIn() {
  Animated.spring(scaleAnim, {
    toValue: 0.96,
    useNativeDriver: true,
    damping: ANIMATION.easing.spring.damping,     // 15
    stiffness: ANIMATION.easing.spring.stiffness, // 120
    isInteraction: false,
  }).start();
}

function handlePressOut() {
  Animated.spring(scaleAnim, {
    toValue: 1.0,
    useNativeDriver: true,
    damping: ANIMATION.easing.spring.damping,     // 15
    stiffness: ANIMATION.easing.spring.stiffness, // 120
    isInteraction: false,
  }).start();
}

// Pressable pattern (HomeFriendCard, HomeWidgetRow):
<Pressable
  onPressIn={handlePressIn}
  onPressOut={handlePressOut}
  onPress={handlePress}
  style={styles.card}  // no pressed callback — remove ({ pressed }) => []
>
  <Animated.View style={{ transform: [{ scale: scaleAnim }] }}>
    {/* card content */}
  </Animated.View>
</Pressable>

// TouchableOpacity pattern (OwnStatusCard, EventCard):
<TouchableOpacity
  onPressIn={handlePressIn}
  onPressOut={handlePressOut}
  activeOpacity={1.0}  // MUST disable opacity fallback
  onPress={onPress}
  style={styles.card}
>
  <Animated.View style={{ transform: [{ scale: scaleAnim }] }}>
    {/* card content */}
  </Animated.View>
</TouchableOpacity>
```
