// uploadAvatar — move verbatim from edit.tsx lines 77-103
async function uploadAvatar(asset: ImagePicker.ImagePickerAsset) {
  if (!session || !asset.base64) return;
  setAvatarLoading(true);
  try {
    const fileExt = asset.uri.split('.').pop()?.toLowerCase() ?? 'jpeg';
    const filePath = `${session.user.id}/avatar.${fileExt}`;
    await supabase.storage.from('avatars').upload(filePath, decode(asset.base64), {
      contentType: `image/${fileExt}`,
      upsert: true,
    });
    const { data: { publicUrl } } = supabase.storage.from('avatars').getPublicUrl(filePath);
    setAvatarUrl(`${publicUrl}?t=${Date.now()}`);
  } catch {
    Alert.alert('Error', "Couldn't upload photo. Make sure the image is under 5MB and try again.");
  } finally {
    setAvatarLoading(false);
  }
}

// handleChangeAvatar — move verbatim from edit.tsx lines 105-141
function handleChangeAvatar() {
  Alert.alert('Change Photo', undefined, [
    {
      text: 'Choose from Library',
      onPress: async () => {
        const result = await ImagePicker.launchImageLibraryAsync({
          mediaTypes: 'images' as ImagePicker.MediaType,
          allowsEditing: true,
          aspect: APP_CONFIG.avatarAspect,
          quality: APP_CONFIG.avatarQuality,
          base64: true,
        });
        if (!result.canceled && result.assets[0]) {
          await uploadAvatar(result.assets[0]);
        }
      },
    },
    {
      text: 'Take Photo',
      onPress: async () => {
        const { status } = await ImagePicker.requestCameraPermissionsAsync();
        if (status !== 'granted') return;
        const result = await ImagePicker.launchCameraAsync({
          mediaTypes: 'images' as ImagePicker.MediaType,
          allowsEditing: true,
          aspect: APP_CONFIG.avatarAspect,
          quality: APP_CONFIG.avatarQuality,
          base64: true,
        });
        if (!result.canceled && result.assets[0]) {
          await uploadAvatar(result.assets[0]);
        }
      },
    },
    { text: 'Cancel', style: 'cancel' },
  ]);
}

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User device → Supabase Storage | Avatar upload bytes cross this boundary; malformed content types or oversized payloads are the primary risk |
| User device → Supabase profiles table | Display name and notification preference writes cross this boundary |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-01-01 | Tampering | uploadAvatar() file path | mitigate | File path is always `{user_id}/avatar.{ext}` derived from authenticated session user ID — user cannot overwrite another user's avatar; Supabase RLS on avatars bucket enforces user_id prefix ownership |
| T-13-01-02 | Information Disclosure | ImagePicker base64 payload | accept | Base64 data stays in-process and is transmitted directly to Supabase Storage over HTTPS; no logging of payload content |
| T-13-01-03 | Spoofing | Avatar content type | mitigate | `contentType: image/${fileExt}` is derived from the URI extension returned by the OS image picker, not from untrusted user input; limited to picker-returned extensions (jpeg, png, heic) |
| T-13-01-04 | Denial of Service | Large avatar upload | accept | APP_CONFIG.avatarQuality caps quality at picker level; 5MB error alert already in uploadAvatar catch block; Supabase Storage has bucket size limits configured separately |
</threat_model>
