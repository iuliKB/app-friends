       ```typescript
       const SWIPE_THRESHOLD_PX = 50; // px â€” short, deliberate swipe; lower than FriendSwipeCard's screen-width-relative threshold since this is a compact calendar, not a full-width card
       const swipeGesture = useMemo(
         () =>
           Gesture.Pan()
             .activeOffsetX([-10, 10])
             .failOffsetY([-15, 15])
             .onEnd((e) => {
               if (Math.abs(e.translationX) < SWIPE_THRESHOLD_PX) return;
               const goNext = e.translationX < 0; // swipe left = next
               if (expanded) {
                 runOnJS(goNext ? onNextMonth : onPrevMonth)();
               } else {
                 runOnJS(goNext ? onNextWeek : onPrevWeek)();
               }
             }),
         [expanded, onNextMonth, onPrevMonth, onNextWeek, onPrevWeek]
       );
       ```
